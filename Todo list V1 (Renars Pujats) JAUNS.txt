  color: gray;

<div class="container mt-5">
<h3 class="text-center mb-4">Todo List V3</h3>
<div class="card">
<div class="card-body">
<div class="input-group mb-3">
<select id="prioritySelect" class="form-select" style="max-width:120px;">
  <option value="secondary">Zema</option>
  <option value="warning" selected>Vidēja</option>
  <option value="danger">SVARĪGI</option>
</select>
<input type="text" id="taskInput" class="form-control" placeholder="Ieraksti darbu...">
<button class="btn btn-primary" id="addBtn">Pievienot</button>
</div>
<input type="text" id="searchInput" class="form-control mb-3" placeholder="Meklēt uzdevumu...">
<div class="btn-group mb-3">
  <button class="btn btn-secondary" data-filter="all">Visi</button>
  <button class="btn btn-outline-secondary" data-filter="active">Aktīvie</button>
  <button class="btn btn-outline-secondary" data-filter="done">Pabeigtie</button>
</div>
<ul id="taskList" class="list-group"></ul>
<div class="d-flex justify-content-between mt-3">
<small id="taskCount" class="text-muted"></small>
<button class="btn btn-sm btn-link text-danger" id="clearDoneBtn">Notīrīt pabeigtos</button>
</div>
</div>
</div>
</div>

<script>
let tasks = [];
let currentFilter = 'all';

const taskList = document.getElementById("taskList");
const taskInput = document.getElementById("taskInput");
const prioritySelect = document.getElementById("prioritySelect");
const searchInput = document.getElementById("searchInput");
const addBtn = document.getElementById("addBtn");
const clearDoneBtn = document.getElementById("clearDoneBtn");
const filterButtons = document.querySelectorAll(".btn-group button");

function saveData(){
  localStorage.setItem("tasks", JSON.stringify(tasks));
}

function loadData(){
  const saved = localStorage.getItem("tasks");
  if(saved) tasks = JSON.parse(saved);
  renderTasks();
}

function renderTasks(){
  taskList.innerHTML = "";
  const query = searchInput.value.toLowerCase();
  let activeCount = 0;

  tasks.forEach((task,index)=>{
    let done = task.done;
    if(currentFilter==='active' && done) return;
    if(currentFilter==='done' && !done) return;
    if(!task.text.toLowerCase().includes(query)) return;

    const li = document.createElement("li");
    li.className = "list-group-item d-flex justify-content-between align-items-center";
    if(done) li.classList.add("izsvitrots");

    const divLeft = document.createElement("div");
    const btnPriority = document.createElement("button");
    btnPriority.className = `btn btn-sm me-2 ${task.priority}`;
    btnPriority.disabled = true;
    btnPriority.innerText = task.priorityText;
    const span = document.createElement("span");
    span.innerText = task.text;
    divLeft.appendChild(btnPriority);
    divLeft.appendChild(span);
    divLeft.style.cursor = "pointer";
    divLeft.addEventListener("click",()=>{ toggleDone(index); });
    
    const divRight = document.createElement("div");
    const btnEdit = document.createElement("button");
    btnEdit.className = "btn btn-info btn-sm me-1";
    btnEdit.innerText = "Labot";
    btnEdit.addEventListener("click",()=>{ editTask(index); });
    const btnDelete = document.createElement("button");
    btnDelete.className = "btn btn-outline-danger btn-sm";
    btnDelete.innerText = "Dzēst";
    btnDelete.addEventListener("click",()=>{ deleteTask(index); });
    divRight.appendChild(btnEdit);
    divRight.appendChild(btnDelete);

    li.appendChild(divLeft);
    li.appendChild(divRight);
    taskList.appendChild(li);

    if(!done) activeCount++;
  });

  document.getElementById("taskCount").innerText = `Atlikuši ${activeCount} darbi`;
}

function addTask(){
  const text = taskInput.value.trim();
  if(!text){ alert("Ievadi uzdevumu!"); return; }
  let priority = prioritySelect.value;
  let priorityText = (priority==='secondary') ? "Zema" : (priority==='warning') ? "Vidēja" : "SVARĪGI";
  tasks.push({ text, priority: `btn-${priority}`, priorityText, done:false });
  taskInput.value = "";
  saveData();
  renderTasks();
}

function toggleDone(index){
  tasks[index].done = !tasks[index].done;
  saveData();
  renderTasks();
}

function editTask(index){
  const newText = prompt("Rediģē uzdevumu:",tasks[index].text);
  if(newText!==null && newText.trim()!==""){
    tasks[index].text = newText;
    saveData();
    renderTasks();
  }
}

function deleteTask(index){
  tasks.splice(index,1);
  saveData();
  renderTasks();
}

function clearDone(){
  tasks = tasks.filter(t=>!t.done);
  saveData();
  renderTasks();
}

filterButtons.forEach(btn=>{
  btn.addEventListener("click",()=>{
    currentFilter = btn.dataset.filter;
    filterButtons.forEach(b=>{
      b.classList.remove("btn-secondary");
      b.classList.add("btn-outline-secondary");
    });
    btn.classList.remove("btn-outline-secondary");
    btn.classList.add("btn-secondary");
    renderTasks();
  });
});

searchInput.addEventListener("input", renderTasks);
addBtn.addEventListener("click", addTask);
clearDoneBtn.addEventListener("click", clearDone);
taskInput.addEventListener("keypress",(e)=>{ if(e.key==="Enter") addTask(); });

loadData();
</script>
</body>
</html>
